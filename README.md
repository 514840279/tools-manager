#### 安装教程

```
# 1. 安装脚手架工具 
npm i -g @vue/cli
npm i -g typescript
npm i -g element-theme 
# 2. 创建一个项目 
vue create manager
cd manager
vue add element-plus

# 3. 使用组件 # node v12
npm i -s vue-router@next 
npm i -s vuex@next
npm i -s axios
npm i -s vue-axios
npm i -s echarts
npm i -s moment  



# 完整引入
import VueRouter from 'vue-router'
Vue.use(VueRouter)

npm run serve
```
